#include <stdio.h>

int main() {
    int a, b, c;

    // Taking input from the user
    printf("Enter three numbers: ");
    scanf("%d %d %d", &a, &b, &c);

    // Finding the greatest number
    if (a >= b && a >= c)
        printf("Greatest number is %d", a);
    else if (b >= a && b >= c)
        printf("Greatest number is %d", b);
    else
        printf("Greatest number is %d", c);

    return 0;
}
